Arrays

A list (container) of items that are organized first to last

0 "red" 1 "blue" 2 "green" 3 "purple"

Creating Arrays

            Array.new() == []

            colours = ["green", "blue", "red", "orange"]

            scores = [60, 43, 93, 78]

            change = [1, "pennies", 2, "dimes", 3, "quarters"]

Accessing Arrays

            colours = ["green", "blue", "red", "orange"]

            colours[0]
            => "green"

            colours[2]
            => "red"

            colours[-1]
? 
              => "orange"

Updating Arrays

            colours = ["green", "blue", "red", "orange"]
            colours[1] = "yellow"
          
            => ["green", "yellow", "red", "orange"]

.append()

            colours = ["green", "blue"]
            colours.append("red")
          
            => ["green", "blue", "red"]

.prepend()

            colours = ["green", "blue"]
            colours.prepend("red")
? 
              => ["red", "green", "blue"]

.insert()

            colours = ["green", "blue", "red"]
            colours.insert(1, "yellow")
? 
              => ["green", "yellow", "blue", "red"]
            
            colours = ["green", "blue", "red"]
            colours.insert(5, "yellow")
? 
              => ["green", "blue", "red", nil, nil, "yellow"]

.each()

            colours = ["green", "blue", "red"]
            colours.each do |colour|
              puts "#{colour} is a colour"
            end
? 
              => "green is a colour"
                 "blue is a colour"
                 "red is a colour"

.each_with_index()

            colours = ["green", "blue", "red"]
            people = ["Kate", "John", "Sally"]

            colours.each_with_index do |colour, index|
              puts "#{people[index]} likes #{colour}"
            end
? 
              => "Kate likes green"
                 "John likes blue"
                 "Sally likes red"

.select()

            scores = [30, 89, 20, 44, 97, 29, 18]
            scores.select {|score| score > 40}
? 
              => [89, 44, 97]
